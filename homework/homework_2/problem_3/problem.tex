\begin{enumerate}
  \item[(a)] For the following ODE, assume $\lambda>0$.\begin{align*}
    \dot{x}&=x(1-x^2)\\

  \end{align*}

    x&\mapsto \lambda x,\\
    y&\mapsto \mu y-x^3.
  \end{align*}
\end{enumerate}
